function onAbilityCast() {
    const caster = Unit.fromEvent();
    const g = new Group();
  
    g.enumUnitsInRange(caster.x, caster.y, 500);
    
    const enemies = g.getUnits().filter((u) => u.isEnemy(caster.owner));
    
    for (const unit of enemies) {
      caster.damageTarget(unit, 50);
    }
  
    g.destroy();
}
